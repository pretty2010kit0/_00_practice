int[] arr = {4, 8, 78, 67, 3, 23};

for(int i=0;i<arr.length-1;i++){    //5轮
	for(int j=0;j<arr.length-1-i;j++){    //每轮比的次数
		if(arr[j]>arr[j+1]){
			int temp = arr[j];
			arr[j] = arr[j+1];
			arr[j+1] = temp;
		}
	}
}

i----------5-i
0轮-----5次
1轮-----4次
2轮-----3次
3轮-----2次
4轮-----1次

   8和78比----不换

   78和3比----换---4,8,67,3,78,23
   78和23比--换---4,8,67,3,23,78

   4和8比

   67和23比------4,8,3,23,67,78
第三轮:------冒出23
   4和8比

   8和23比

   4和3比---------3,4,8,23,67,78
   4和8比
